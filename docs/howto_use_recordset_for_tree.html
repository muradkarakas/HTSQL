How to: use Recordset variable and to_json function for populating a tree node.

In order to generate json data from database, record set variable and to_json function can be used as shown in the example below,

Record set must ;

  • have at least 3 columns: "id", "text", "parent".
  • If actual table column name is different from "id", "text" and "parent", column alias can be used as showed above.

Record set can optionally have;

  • "children" column.
    As stated in jsTree documentation, recorset's column value must be either "true" or "false". True means the node has child(s) node, false means node is a leaf node and cannot be expanded.

67 char cbTree.data.tree_node_expanded(char node_id, char node_type) {
68  char json;
69 
70  /* message('node_id : ' || node_id || '. node_type : ' || node_type); */
71 
72  if (node_id == '#') then
73 
74  rs = select
75  region_id as id,
76  region_name as text,
77  '#' as parent,
78  'R' as type,
79  case (select count(*) from htsql.countries c where c.region_id = r.region_id)
80  when 0 then
81  'false'
82  else
83  'true'
84  end children
85  from
86  htsql.regions r;
87  if (rs) then
88  /* */
89  json = to_json(rs);
90  else
91  json = '[ { "id" : "no_region", "parent" : "#", "text" : "Click here to add new region" } ]';
92  end if;
93  end if;
94 
95  if (node_type == 'R') then
96  rs = select
97  country_id as id,
98  country_name as text,
99  cc.region_id as parent,
100  'C' as type,
101  case (select count(*) from htsql.countries c where c.region_id = cc.region_id)
102  when 0 then
103  'false'
104  else
105  'true'
106  end children
107 
108  from
109  htsql.countries cc
110  where
111  cc.region_id = :node_id;
112  json = to_json(rs);
113  end if;
114 
115  if (node_type == 'C') then
116 
117  rs = select
118  PROVINCE_ID as id,
119  PROVINCE_name as text,
120  country_id as parent,
121  'P' as type,
122  case (select count(*) from htsql.counties c where c.province_id = cc.province_id)
123  when 0 then
124  'false'
125  else
126  'true'
127  end children
128  from
129  htsql.provinces cc
130  where
131  cc.country_id = :node_id;
132 
133  json = to_json(rs);
134  end if;
135 
136  if (node_type == 'P') then
137 
138  rs = select
139  county_id as id,
140  county_name as text,
141  province_id as parent,
142  'false' as children
143  from
144  htsql.counties c
145  where
146  c.province_id = :node_id;
147 
148  json = to_json(rs);
149  end if;
150 
151  return json;
152 }
